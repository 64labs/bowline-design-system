# Theming

The Bowline theme config allows you to create or modify the token values used throughout the system. Typography, color, spacing, and nearly every other aspect of the system can be customized to suit your project's specifications.

```js
// bowline.config.js

export default {
  // a base multiple for spacing and layout
  grid: 4,

  // map of your project's breakpoints
  screens: {
    sm: 480,
    md: 640,
    lg: 960,
  },

  // typeface tokens for text and heading
  fontFamily: {
    text: '"Open Sans", -apple-system, system-ui, sans-serif',
    heading: '"Open Sans", -apple-system, system-ui, sans-serif',
  },

  // font-specific scaling factors used for baseline cropping
  fontScale: {
    text: {
      descenderHeightScale: 0.1155,
      capHeight: 0.725,
    },
    heading: {
      descenderHeightScale: 0.1155,
      capHeight: 0.725,
    },
  },

  // spacing tokens for layout
  spacing: {
    none: 0,
    xxsmall: grid * 1,
    xsmall: grid * 2,
    smallish: grid * 3,
    small: grid * 4,
    gutter: grid * 6,
    medium: grid * 8,
    large: grid * 9,
    largeish: grid * 10,
    xlarge: grid * 12,
    xxlarge: grid * 16,
    half: "50%",
    full: "100%",
    fullvh: "100vh",
    fullvw: "100vw",
  },

  // tokens for <ContentBlock> `width` prop
  contentWidth: {
    medium: 640,
    large: 720,
    max: 1280,
  },

  colors: {
    // base colors
    brand: "#54494B",
    brandLight: "#D3D5D4",
    brandAccent: "#7F6A93",
    positive: "green",
    positiveLight: "#EEF4EF",
    critical: "#f44336",
    info: "#9DB5B2",
    infoLight: "#1e468c",
    promote: "#7F6A93",
    promoteLight: "#efeaf3",
    neutral: "#1c1c1c",
    neutralInverted: "#ffffff",
    secondary: "#878E99",
    secondaryInverted: "#ffffff",
    subtle: "#eee",
    black: "#1c1c1c",
    white: "#fff",

    // applied colors
    formAccent: colors.brandAccent,
    focus: colors.brand,
    link: colors.brand,
    linkVisited: colors.brand,
  },

  // tokens for <Text> `size` prop, expects a default,
  // optional sizes per theme screens (breakpoints)
  textSizes: {
    xsmall: {
      default: {
        size: 12, // font-size in pixels
        rows: 4, // line-height in grid multiples (grid * 4 = 16px)
      },
      sm: {
        size: 12,
        rows: 4,
      },
    },
  },

  // same as `textSizes`, applied when <Text> prop `heading` is true
  headingSizes: {
    xsmall: {
      default: {
        size: 14,
        rows: 5,
      },
      sm: {
        size: 14,
        rows: 5,
      },
    },
  },

  // tokens for <Text> `weight` prop
  fontWeights: {
    regular: 400,
    medium: 600,
    strong: 700,
  },

  // tokens for <Text> 'kerning' prop
  letterSpacing: {
    small: "0.025em",
    medium: "0.05em",
    large: "0.075em",
  },

  // shadow styles, also used for fake borders
  shadows: {
    borderStandard: {
      style: `0px 0px 0px 2px inset`,
      color: colors.promoteLight,
      colorInverted: "white",
    },
  },

  // tokens for border-related props
  border: {
    radius: {
      standard: "3px",
      medium: "6px",
      large: "9px",
      half: "50%",
    },
    style: {
      none: 0,
      standard: "1px solid",
      double: "2px solid",
    },
  },
}
```
