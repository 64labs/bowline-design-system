import {
  Image,
  TransitionImage,
  Button,
  Columns,
} from "@64labs/bowline-design-system"

export const component = Image

export const Example = () => {
  const set1 = [101, 102, 104]
  const set2 = [106, 108, 109]
  const sets = { set1, set2 }
  const [selectedSet, setSelectedSet] = React.useState("set1")
  return (
    <Stack>
      <Button
        weight="weak"
        onClick={() => setSelectedSet(selectedSet === "set1" ? "set2" : "set1")}
      >
        Change images
      </Button>
      <Columns cols={3}>
        {sets[selectedSet].map((id, idx) => (
          <TransitionImage
            key={idx}
            src={`https://picsum.photos/id/${id}/400/500`}
            width={4}
            height={5}
            fluid
          />
        ))}
      </Columns>
    </Stack>
  )
}
